`AUX` nvarchar(255),
`TIPOLOGIA` nvarchar(255),
`NOMBRE` nvarchar(255),
`MUNICIPIO` nvarchar(255),
`RESPONSABLE` nvarchar(255),
`DOMICILIO` nvarchar(255),
`TELEFONO` nvarchar(255),
`RED INSTITUCIONAL` nvarchar(255),
`NUMERO DE CELULAR` nvarchar(255),

)
